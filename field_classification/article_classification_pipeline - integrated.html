Two-Step Discipline Classification Using LLMs

Author

Adapted from German Federal Statistical Office (Stabu) Classification

Published

December 3, 2025

Overview

This pipeline classifies scientific articles into disciplinary clusters and fields based on title, abstract, and journal title. The classification schema is adapted from the German Federal Statistical Office (Fächersystematik) with modifications.

Key Design Decisions

  1. Two-step classification: First assign to a cluster, then (if cluster has >50 papers) assign to a field within that cluster
  2. Single-label assignment: Each article receives exactly one primary classification
  3. Content-first classification: Classify based on substantive content, using journal title as secondary disambiguation
  4. Hierarchical flexibility: Clusters provide broad categories; fields provide granularity where needed

Modifications from Original Stabu Schema

  • Merged all theology into “Religious Studies & Theology”

  • Merged Media Studies into Communication Studies

  • Added: Marketing, Behavioural/Experimental Economics, Experimental Philosophy

  • Tightened definition of Occupational/Organisational/Business Psychology

  • Changed General and Cognitive Psychology to Cognitive Psychology, to distinguish from Psychology (General)

  • Pull Science Studies from Philosophy into its own cluster to allow for correct classification of metascience papers

  • Added specific rules for metascience, consumer behaviour, and JDM research

clean_json_text <- function(x) {
  x <- as.character(x)
  # Build replacements safely
  bytes <- c(0x91, 0x92, 0x93, 0x94, 0x96, 0x97, 0x85, 0xD5, 0xCA)
  repl <- c("'", "'", '"', '"', "-", "-", "...", "'", "")
  names(repl) <- vapply(
    bytes,
    function(b) rawToChar(as.raw(b), multiple = FALSE),
    FUN.VALUE = character(1)
  )
  # Replace byte patterns
  for (bad in names(repl)) {
    x <- gsub(bad, repl[[bad]], x, fixed = TRUE, useBytes = TRUE)
  }
  # Drop control characters except newline and tab
  x <- stringi::stri_replace_all_regex(
    x,
    "[\\p{Cc}&&[^\\n\\t]]",
    ""
  )
  # Final UTF-8 normalisation
  x <- iconv(x, from = "", to = "UTF-8", sub = "'")
  x
}

clean_abstract <- function(text) {
  text <- str_remove_all(text, "Copyright ©.*$") # Remove 'Copyright ©' to the end of the string
  text <- str_remove_all(text, "©.*$") # Remove '©' to the end of the string
  text <- str_remove_all(text, "\\(PsycINFO Database Record \\(c\\) \\d{4} APA, all rights reserved\\)") # Remove PsycINFO notice
  text <- str_remove_all(text, "<[^>]+>") # Remove HTML tags
  return(text)
}



# Helper function for missing value handling
`%||%` <- function(x, y) {
  if (is.null(x)) return(y)
  if (is.character(x) && length(x) == 1 && is.na(x)) return(y)
  if (length(x) == 0) return(y)
  x
}

# Validate cluster schema structure
validate_cluster_schema <- function(x) {
  if (!is.list(x) || is.null(x$clusters)) {
    stop("cluster_schema must be a list with element 'clusters'.")
  }

  allowed_top <- c("version", "description", "clusters")
  extra_top <- setdiff(names(x), allowed_top)
  if (length(extra_top) > 0) {
    warning(
      "Ignoring unsupported top-level fields in cluster_schema: ",
      paste(extra_top, collapse = ", ")
    )
  }

  x$clusters <- lapply(x$clusters, function(cl) {
    if (is.null(cl$id) || is.null(cl$label)) {
      stop("Each cluster must have 'id' and 'label'.")
    }
    allowed <- c("id", "label", "notes")
    extra <- setdiff(names(cl), allowed)
    if (length(extra) > 0) {
      warning(
        "Ignoring unsupported fields in cluster '", cl$id, "': ",
        paste(extra, collapse = ", ")
      )
    }
    # Don't add empty fields - code handles missing fields with %||% operator
    cl
  })

  x
}

      "id", "label", "description",
      "positive_criteria", "exclusions"

      warning(

        paste(extra, collapse = ", ")
      )

    f
  })

  x

Cluster Classification Prompt Template

# Reference template - used with glue(cluster_prompt_template, cluster_schema = cluster_schema_json)
cluster_prompt_template <- ('
You classify scientific articles into a single primary discipline cluster.

Use ONLY the clusters defined in the JSON schema below.
Choose the cluster that best reflects the article\'s main research question and methods.
Often, the journal title specifies the field, but make sure to check that it matches the article content.
For interdisciplinary journals, focus on the article content (title and abstract).
If an article is interdisciplinary, pick the cluster that best fits the core contribution.
Never invent new clusters; always return one of the existing id values.

Key distinctions:
- Psychology vs Sociology: Individual processes vs social structure/institutions
- Psychology vs Economics: Psychological theory vs economic framing (utility, incentives, markets)
- Education vs Teaching: Educational systems vs teaching methods in a discipline
- Media vs Political Science: Media/communication vs political processes/institutions
- Metascience: Only for science in general, not discipline-specific methods
- Methods papers: Belong to the discipline being studied

CLUSTER SCHEMA (JSON):
{cluster_schema}

You will receive articles as JSON with: article_title, article_abstract, journal_title
')

Psychology Fields Classification Prompt Template

# Reference template - used with glue(psychology_field_prompt_template, psychology_fields_schema = psychology_fields_json)
psychology_field_prompt_template <- '
You classify psychology articles into a single primary psychology subfield.

Use ONLY the fields defined in the JSON schema below.
Choose the subfield that best reflects the article\'s main theoretical framing.
Prioritise article content (title and abstract). Use journal title only to resolve ambiguity.
If interdisciplinary, pick the subfield that best fits the core psychological contribution.
Never invent new fields; always return one of the existing id values.

Key distinctions:
- Psychology (General): Last resort only when no single subfield clearly dominates
- Occupational Psychology: Requires BOTH work-specific constructs AND theoretical claims about work
  - Workplace as setting only → classify by underlying theory (e.g., prejudice reduction → Social)
  - Work-specific constructs (leadership, burnout, job satisfaction) → Occupational
- Educational Psychology: Learning/motivation/assessment in educational settings only
- Methods: Primary contribution must be methodological (new measure, technique, or meta-analytic method)

PSYCHOLOGY FIELDS SCHEMA (JSON):
{psychology_fields_schema}

You will receive articles as JSON with: article_title, article_abstract, journal_title
'

# Create cluster classification chat with provider flexibility
create_cluster_chat <- function(
  model = "gpt-5.1",
  provider = "openai",
  api_args = list(temperature = 0)
) {
  system_prompt <- glue(cluster_prompt_template,
                        cluster_schema = cluster_schema_json)

  if (provider == "openai") {
    chat_openai(
      system_prompt = system_prompt,
      model = model,
      api_args = api_args,
      echo = "none"
    )
  } else if (provider == "anthropic") {
    chat_anthropic(
      system_prompt = system_prompt,
      model = model,
      api_args = api_args,
      echo = "none"
    )
  } else {
    stop("Provider must be 'openai' or 'anthropic'")
  }
}

Clusters (Batch Processing)

classify_articles_batch <- function(
  articles_df,
  title_col = "title",
  abstract_col = "abstract",
  journal_col = "journal",
  model = "gpt-5.1",
  provider = "openai"
) {
  # Create chat for cluster classification
  cluster_chat <- create_cluster_chat(
    model = model,
    provider = provider,
    api_args = list(temperature = 0)
  )

  # Prepare prompts for batch processing
  articles_list <- articles_df |>
    rowwise() |>
    reframe(
      article = toJSON(
        list(
          article_title = .data[[title_col]] %||% "",
          article_abstract = .data[[abstract_col]] %||% "",
          journal_title = .data[[journal_col]] %||% ""
        ),
        auto_unbox = TRUE
      )
    ) |>
    pull(article) |>
    as.character()  |>
    clean_json_text() |>
    as.list()

  # Batch classify clusters using parallel processing
  cat("Classifying articles into clusters (parallel processing)...\n")
  cluster_results_list <- parallel_chat_structured(
    chat = cluster_chat,
    prompts = articles_list,
    type = type_cluster_result
  )
  
  # Combine results with original data
  articles_df |>
    bind_cols(cluster_results_list)
}

Field Classification for Large Clusters

Code 
classify_cluster_fields <- function(
  cluster_results,
  cluster_breakdowns,
  title_col = "title",
  abstract_col = "abstract",
  journal_col = "journal",
  model = "gpt-5.1",
  provider = "openai",
  field_threshold = 50
) {
  # Count articles per cluster
  cluster_counts <- cluster_results |>
    count(cluster_id) |>
    arrange(desc(n))

  cat("\n=== CLUSTERS REQUIRING FIELD CLASSIFICATION ===\n")
  clusters_above_threshold <- cluster_counts |>
    filter(n >= field_threshold)

  if (nrow(clusters_above_threshold) > 0) {
    print(clusters_above_threshold)
  } else {
    cat("No clusters with >= ", field_threshold, " articles\n")
    return(cluster_results)
  }

  # Process each cluster with available field schema
  results_with_fields <- tibble()

  for (cluster_id in clusters_above_threshold$cluster_id) {
    if (!(cluster_id %in% names(cluster_breakdowns))) {
      cat(sprintf("Note: No field schema available for cluster '%s'\n", cluster_id))
      next
    }

    config <- cluster_breakdowns[[cluster_id]]
    articles_idx <- which(cluster_results$cluster_id == cluster_id)
    n_articles <- length(articles_idx)

    cat(sprintf(
      "\nClassifying %d %s articles into fields (parallel processing)...\n",
      n_articles, cluster_id
    ))

    # Create field chat for this cluster
    field_chat <- config$chat_constructor(
      model = model,
      provider = provider,
      api_args = list(temperature = 0)
    )

    # Prepare prompts
    field_prompts <- cluster_results[articles_idx, ] |>
      rowwise() |>
      reframe(
        article = toJSON(
          list(
            article_title = .data[[title_col]] %||% "",
            article_abstract = .data[[abstract_col]] %||% "",
            journal_title = .data[[journal_col]] %||% ""
          ),
          auto_unbox = TRUE
        )
      ) |>
      pull(article) |>
      as.character() |>
      clean_json_text() |>
      as.list()

    # Classify fields
    field_results_list <- parallel_chat_structured(
      chat = field_chat,
      prompts = field_prompts,
      type = type_field_result
    )
    
    current_cluster <- cluster_id
    
    current_results <- cluster_results %>% 
      filter(cluster_id == current_cluster) %>% 
      bind_cols(field_results_list)
    
    cluster_results <- cluster_results %>% 
      filter(cluster_id != current_cluster)
    
    results_with_fields <- bind_rows(results_with_fields, current_results)

  }

  bind_rows(results_with_fields, cluster_results)
}

## Overview

This pipeline classifies scientific articles into disciplinary clusters and fields based on title, abstract, and journal title. The classification schema is adapted from the German Federal Statistical Office (Fächersystematik) with modifications.

### Key Design Decisions

1. **Two-step classification**: First assign to a cluster, then (if cluster has >50 papers) assign to a field within that cluster
2. **Single-label assignment**: Each article receives exactly one primary classification
3. **Content-first classification**: Classify based on substantive content, using journal title as secondary disambiguation
4. **Hierarchical flexibility**: Clusters provide broad categories; fields provide granularity where needed

### Modifications from Original Stabu Schema

- Merged all theology into "Religious Studies & Theology"
- Merged Media Studies into Communication Studies
- Added: Marketing, Behavioural/Experimental Economics, Experimental Philosophy
- Tightened definition of Occupational/Organisational/Business Psychology
- Changed General and Cognitive Psychology to Cognitive Psychology, to distinguish from Psychology (General)
- Pull Science Studies from Philosophy into its own cluster to allow for correct classification of metascience papers

- Added specific rules for metascience, consumer behaviour, and JDM research

```{r}
  x <- as.character(x)
  # Build replacements safely
  bytes <- c(0x91, 0x92, 0x93, 0x94, 0x96, 0x97, 0x85, 0xD5, 0xCA)
  repl <- c("'", "'", '"', '"', "-", "-", "...", "'", "")
  names(repl) <- vapply(
    bytes,
    function(b) rawToChar(as.raw(b), multiple = FALSE),
    FUN.VALUE = character(1)
  )
  # Replace byte patterns
  for (bad in names(repl)) {
    x <- gsub(bad, repl[[bad]], x, fixed = TRUE, useBytes = TRUE)
  }
  # Drop control characters except newline and tab
  x <- stringi::stri_replace_all_regex(
    x,
    "[\\p{Cc}&&[^\\n\\t]]",
    ""
  )
  # Final UTF-8 normalisation
  x <- iconv(x, from = "", to = "UTF-8", sub = "'")
  x
}

clean_abstract <- function(text) {
  text <- str_remove_all(text, "Copyright ©.*$") # Remove 'Copyright ©' to the end of the string
  text <- str_remove_all(text, "©.*$") # Remove '©' to the end of the string
  text <- str_remove_all(text, "\\(PsycINFO Database Record \\(c\\) \\d{4} APA, all rights reserved\\)") # Remove PsycINFO notice
  text <- str_remove_all(text, "<[^>]+>") # Remove HTML tags
  return(text)
}



# Helper function for missing value handling
`%||%` <- function(x, y) {
  if (is.null(x)) return(y)
  if (is.character(x) && length(x) == 1 && is.na(x)) return(y)
  if (length(x) == 0) return(y)
  x
}
```

```{r}
# Validate cluster schema structure
validate_cluster_schema <- function(x) {
  if (!is.list(x) || is.null(x$clusters)) {
    stop("cluster_schema must be a list with element 'clusters'.")
  }

  allowed_top <- c("version", "description", "clusters")
  extra_top <- setdiff(names(x), allowed_top)
  if (length(extra_top) > 0) {
    warning(
      "Ignoring unsupported top-level fields in cluster_schema: ",
      paste(extra_top, collapse = ", ")
    )
  }

  x$clusters <- lapply(x$clusters, function(cl) {
    if (is.null(cl$id) || is.null(cl$label)) {
      stop("Each cluster must have 'id' and 'label'.")
    }
    allowed <- c("id", "label", "notes")
    extra <- setdiff(names(cl), allowed)
    if (length(extra) > 0) {
      warning(
        "Ignoring unsupported fields in cluster '", cl$id, "': ",
        paste(extra, collapse = ", ")
      )
    }
    # Don't add empty fields - code handles missing fields with %||% operator
    cl
  })

  x
}
```

```{r}
#| label: batch-processing-parallel

classify_articles_batch <- function(
  articles_df,
  title_col = "title",
  abstract_col = "abstract",
  journal_col = "journal",
  model = "gpt-5.1",
  provider = "openai"
) {
  # Create chat for cluster classification
  cluster_chat <- create_cluster_chat(
    model = model,
    provider = provider,
    api_args = list(temperature = 0)
  )

  # Prepare prompts for batch processing
  articles_list <- articles_df |>
    rowwise() |>
    reframe(
      article = toJSON(
        list(
          article_title = .data[[title_col]] %||% "",
          article_abstract = .data[[abstract_col]] %||% "",
          journal_title = .data[[journal_col]] %||% ""
        ),
        auto_unbox = TRUE
      )
    ) |>
    pull(article) |>
    as.character()  |>
    clean_json_text() |>
    as.list()

  # Batch classify clusters using parallel processing
  cat("Classifying articles into clusters (parallel processing)...\n")
  cluster_results_list <- parallel_chat_structured(
    chat = cluster_chat,
    prompts = articles_list,
    type = type_cluster_result
  )
  
  # Combine results with original data
  articles_df |>
    bind_cols(cluster_results_list)
}
```

## Field Classification for Large Clusters

```{r}
#| label: field-processing-parallel

classify_cluster_fields <- function(
  cluster_results,
  cluster_breakdowns,
  title_col = "title",
  abstract_col = "abstract",
  journal_col = "journal",
  model = "gpt-5.1",
  provider = "openai",
  field_threshold = 50
) {
  # Count articles per cluster
  cluster_counts <- cluster_results |>
    count(cluster_id) |>
    arrange(desc(n))

  cat("\n=== CLUSTERS REQUIRING FIELD CLASSIFICATION ===\n")
  clusters_above_threshold <- cluster_counts |>
    filter(n >= field_threshold)

  if (nrow(clusters_above_threshold) > 0) {
    print(clusters_above_threshold)
  } else {
    cat("No clusters with >= ", field_threshold, " articles\n")
    return(cluster_results)
  }

  # Process each cluster with available field schema
  results_with_fields <- tibble()

  for (cluster_id in clusters_above_threshold$cluster_id) {
    if (!(cluster_id %in% names(cluster_breakdowns))) {
      cat(sprintf("Note: No field schema available for cluster '%s'\n", cluster_id))
      next
    }

    config <- cluster_breakdowns[[cluster_id]]
    articles_idx <- which(cluster_results$cluster_id == cluster_id)
    n_articles <- length(articles_idx)

    cat(sprintf(
      "\nClassifying %d %s articles into fields (parallel processing)...\n",
      n_articles, cluster_id
    ))

    # Create field chat for this cluster
    field_chat <- config$chat_constructor(
      model = model,
      provider = provider,
      api_args = list(temperature = 0)
    )

    # Prepare prompts
    field_prompts <- cluster_results[articles_idx, ] |>
      rowwise() |>
      reframe(
        article = toJSON(
          list(
            article_title = .data[[title_col]] %||% "",
            article_abstract = .data[[abstract_col]] %||% "",
            journal_title = .data[[journal_col]] %||% ""
          ),
          auto_unbox = TRUE
        )
      ) |>
      pull(article) |>
      as.character() |>
      clean_json_text() |>
      as.list()

    # Classify fields
    field_results_list <- parallel_chat_structured(
      chat = field_chat,
      prompts = field_prompts,
      type = type_field_result
    )
    
    current_cluster <- cluster_id
    
    current_results <- cluster_results %>% 
      filter(cluster_id == current_cluster) %>% 
      bind_cols(field_results_list)
    
    cluster_results <- cluster_results %>% 
      filter(cluster_id != current_cluster)
    
    results_with_fields <- bind_rows(results_with_fields, current_results)

  }

  bind_rows(results_with_fields, cluster_results)
}
```
