# ellmer::parallel_chat uses httr2::req_perform_parallel which has a cli progress
# bar that breaks inside knitr. Override classify functions to process sequentially.
classify_articles_sequential <- function(
  articles_df,
  title_col = "title",
  abstract_col = "abstract",
  journal_col = "journal",
  model = "openai/gpt-oss-120b",
  provider = "deepinfra"
) {
  articles_list <- articles_df |>
    rowwise() |>
    reframe(
      article = toJSON(
        list(
          article_title = .data[[title_col]] %||% "",
          article_abstract = .data[[abstract_col]] %||% "",
          journal_title = .data[[journal_col]] %||% ""
        ),
        auto_unbox = TRUE
      )
    ) |>
    pull(article) |>
    as.character() |>
    clean_json_text()

  n <- length(articles_list)
  cluster_ids <- character(n)
  confidences <- numeric(n)
  secondary_ids <- character(n)
  secondary_confs <- numeric(n)
  reasonings <- character(n)

  cat(sprintf("Classifying %d articles into clusters (sequential)...\n", n))

  for (i in seq_len(n)) {
    if (i %% 10 == 0 || i == n) cat(sprintf("  %d / %d\n", i, n))

    chat <- create_cluster_chat(model = model, provider = provider,
                                api_args = list(temperature = 0))
    tryCatch({
      chat$chat(articles_list[i])
      turns <- chat$get_turns()
      assistant_turn <- turns[[2]]

      content <- assistant_turn@text
      parsed <- tryCatch(fromJSON(content),
        error = function(e) list(cluster_id = NA_character_, confidence = NA_real_,
                                 secondary_cluster_id = NA_character_,
                                 secondary_confidence = NA_real_))

      reasoning <- tryCatch(
        assistant_turn@json$choices[[1]]$message$reasoning_content %||% NA_character_,
        error = function(e) NA_character_)

      cluster_ids[i] <- parsed$cluster_id %||% NA_character_
      confidences[i] <- parsed$confidence %||% NA_real_
      secondary_ids[i] <- parsed$secondary_cluster_id %||% NA_character_
      secondary_confs[i] <- parsed$secondary_confidence %||% NA_real_
      reasonings[i] <- reasoning
    }, error = function(e) {
      message("  Error on article ", i, ": ", e$message)
      cluster_ids[i] <<- NA_character_
      confidences[i] <<- NA_real_
      secondary_ids[i] <<- NA_character_
      secondary_confs[i] <<- NA_real_
      reasonings[i] <<- NA_character_
    })
  }

  articles_df |>
    bind_cols(tibble(
      cluster_id = cluster_ids,
      confidence = confidences,
      secondary_cluster_id = secondary_ids,
      secondary_confidence = secondary_confs,
      reasoning = reasonings
    ))
}

classify_fields_sequential <- function(
  cluster_results,
  cluster_breakdowns,
  title_col = "title",
  abstract_col = "abstract",
  journal_col = "journal",
  model = "openai/gpt-oss-120b",
  provider = "deepinfra",
  field_threshold = 100
) {
  cluster_counts <- cluster_results |> count(cluster_id) |> arrange(desc(n))

  cat("\n=== CLUSTERS REQUIRING FIELD CLASSIFICATION ===\n")
  clusters_above_threshold <- cluster_counts |> filter(n >= field_threshold)

  if (nrow(clusters_above_threshold) > 0) {
    print(clusters_above_threshold)
  } else {
    cat("No clusters with >= ", field_threshold, " articles\n")
    return(cluster_results)
  }

  results_with_fields <- tibble()

  for (cid in clusters_above_threshold$cluster_id) {
    if (!(cid %in% names(cluster_breakdowns))) {
      cat(sprintf("Note: No field schema available for cluster '%s'\n", cid))
      next
    }

    config <- cluster_breakdowns[[cid]]
    articles_idx <- which(cluster_results$cluster_id == cid)
    n_articles <- length(articles_idx)
    cat(sprintf("\nClassifying %d %s articles into fields (sequential)...\n",
                n_articles, cid))

    field_ids <- character(n_articles)
    field_confs <- numeric(n_articles)
    field_reasonings <- character(n_articles)

    for (j in seq_len(n_articles)) {
      idx <- articles_idx[j]
      if (j %% 10 == 0 || j == n_articles) cat(sprintf("  %d / %d\n", j, n_articles))

      article_json <- toJSON(list(
        article_title = cluster_results[[title_col]][idx] %||% "",
        article_abstract = cluster_results[[abstract_col]][idx] %||% "",
        journal_title = cluster_results[[journal_col]][idx] %||% ""
      ), auto_unbox = TRUE) |> as.character() |> clean_json_text()

      field_chat <- config$chat_constructor(model = model, provider = provider,
                                            api_args = list(temperature = 0))
      tryCatch({
        field_chat$chat(article_json)
        turns <- field_chat$get_turns()
        assistant_turn <- turns[[2]]

        parsed <- tryCatch(fromJSON(assistant_turn@text),
          error = function(e) list(field_id = NA_character_, confidence = NA_real_))

        reasoning <- tryCatch(
          assistant_turn@json$choices[[1]]$message$reasoning_content %||% NA_character_,
          error = function(e) NA_character_)

        field_ids[j] <- parsed$field_id %||% NA_character_
        field_confs[j] <- parsed$confidence %||% NA_real_
        field_reasonings[j] <- reasoning
      }, error = function(e) {
        message("  Error on field article ", j, ": ", e$message)
        field_ids[j] <<- NA_character_
        field_confs[j] <<- NA_real_
        field_reasonings[j] <<- NA_character_
      })
    }

    current_cluster <- cid
    current_results <- cluster_results |>
      filter(cluster_id == current_cluster) |>
      bind_cols(tibble(
        field_id = field_ids,
        field_confidence = field_confs,
        field_reasoning = field_reasonings
      ))
    cluster_results <- cluster_results |> filter(cluster_id != current_cluster)
    results_with_fields <- bind_rows(results_with_fields, current_results)
  }

  bind_rows(results_with_fields, cluster_results)
}
